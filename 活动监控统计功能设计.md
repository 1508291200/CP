### 1. 鼠标监听器

**Windows平台（使用 node-global-key-listener 或 iohook）**

```typescript
import { GlobalKeyboardListener } from 'node-global-key-listener';
// 或者使用 iohook

class MouseTracker {
  private clickCount = {
    left: 0,
    right: 0,
    middle: 0,
  };
  private moveDistance = 0;
  private lastPosition = { x: 0, y: 0 };
  
  start(): void {
    // 使用 iohook 监听全局鼠标事件
    const ioHook = require('iohook');
    
    // 监听鼠标点击
    ioHook.on('mouseclick', (event: any) => {
      switch (event.button) {
        case 1: // 左键
          this.clickCount.left++;
          break;
        case 2: // 右键
          this.clickCount.right++;
          break;
        case 3: // 中键
          this.clickCount.middle++;
          break;
      }
      
      this.notifyUpdate();
    });
    
    // 监听鼠标移动
    ioHook.on('mousemove', (event: any) => {
      const { x, y } = event;
      
      if (this.lastPosition.x !== 0) {
        const distance = Math.sqrt(
          Math.pow(x - this.lastPosition.x, 2) +
          Math.pow(y - this.lastPosition.y, 2)
        );
        this.moveDistance += distance;
      }
      
      this.lastPosition = { x, y };
    });
    
    // 启动监听
    ioHook.start();
  }
  
  stop(): void {
    require('iohook').stop();
  }
  
  getStats(): MouseStats {
    return {
      clicks: { ...this.clickCount },
      totalClicks: Object.values(this.clickCount).reduce((a, b) => a + b, 0),
      moveDistance: Math.round(this.moveDistance),
      moveDistanceKm: (this.moveDistance / 1000000).toFixed(2), // 转换为公里
    };
  }
  
  reset(): void {
    this.clickCount = { left: 0, right: 0, middle: 0 };
    this.moveDistance = 0;
  }
  
  private notifyUpdate(): void {
    // 通知渲染进程更新
    BrowserWindow.getAllWindows()[0]?.webContents.send(
      'activity:update',
      { type: 'mouse', data: this.getStats() }
    );
  }
}

interface MouseStats {
  clicks: {
    left: number;
    right: number;
    middle: number;
  };
  totalClicks: number;
  moveDistance: number;
  moveDistanceKm: string;
}
```

### 2. 键盘监听器

```typescript
class KeyboardTracker {
  private keyPressCount = 0;
  private characterCount = 0;
  private recentKeys: string[] = [];
  private keyFrequency: Map<string, number> = new Map();
  
  start(): void {
    const ioHook = require('iohook');
    
    ioHook.on('keydown', (event: any) => {
      this.keyPressCount++;
      
      // 记录按键频率
      const keyCode = event.keycode;
      const count = this.keyFrequency.get(keyCode) || 0;
      this.keyFrequency.set(keyCode, count + 1);
      
      // 统计有效字符（排除功能键）
      if (this.isCharacterKey(event)) {
        this.characterCount++;
      }
      
      // 记录最近按键（用于分析输入模式）
      this.recentKeys.push(this.getKeyName(event.keycode));
      if (this.recentKeys.length > 100) {
        this.recentKeys.shift();
      }
      
      this.notifyUpdate();
    });
    
    ioHook.start();
  }
  
  stop(): void {
    require('iohook').stop();
  }
  
  getStats(): KeyboardStats {
    return {
      totalKeys: this.keyPressCount,
      characters: this.characterCount,
      wordsEstimate: Math.round(this.characterCount / 5), // 平均5个字符一个单词
      topKeys: this.getTopKeys(10),
      keysPerMinute: this.calculateKPM(),
    };
  }
  
  reset(): void {
    this.keyPressCount = 0;
    this.characterCount = 0;
    this.keyFrequency.clear();
  }
  
  private isCharacterKey(event: any): boolean {
    const keycode = event.keycode;
    // 字母、数字、标点符号
    return (
      (keycode >= 65 && keycode <= 90) ||  // A-Z
      (keycode >= 48 && keycode <= 57) ||  // 0-9
      (keycode >= 186 && keycode <= 222)   // 标点符号
    );
  }
  
  private getTopKeys(limit: number): Array<{ key: string; count: number }> {
    const sorted = Array.from(this.keyFrequency.entries())
      .sort((a, b) => b[1] - a[1])
      .slice(0, limit);
    
    return sorted.map(([keycode, count]) => ({
      key: this.getKeyName(parseInt(keycode)),
      count,
    }));
  }
  
  private getKeyName(keycode: number): string {
    // 映射keycode到按键名称
    const keyMap: Record<number, string> = {
      65: 'A', 66: 'B', 67: 'C', // ...
      // 完整映射表
    };
    return keyMap[keycode] || `Key${keycode}`;
  }
  
  private calculateKPM(): number {
    // 计算每分钟按键数（需要结合时间追踪）
    // 这里简化实现
    return 0;
  }
  
  private notifyUpdate(): void {
    BrowserWindow.getAllWindows()[0]?.webContents.send(
      'activity:update',
      { type: 'keyboard', data: this.getStats() }
    );
  }
}

interface KeyboardStats {
  totalKeys: number;
  characters: number;
  wordsEstimate: number;
  topKeys: Array<{ key: string; count: number }>;
  keysPerMinute: number;
}
```

### 3. 应用使用追踪器

```typescript
import { activeWindow } from 'active-win';

class ApplicationTracker {
  private currentApp: string | null = null;
  private appUsage: Map<string, AppUsageData> = new Map();
  private checkInterval: NodeJS.Timeout | null = null;
  
  start(): void {
    // 每秒检查一次当前活跃窗口
    this.checkInterval = setInterval(() => {
      this.checkActiveWindow();
    }, 1000);
  }
  
  stop(): void {
    if (this.checkInterval) {
      clearInterval(this.checkInterval);
    }
  }
  
  private async checkActiveWindow(): Promise<void> {
    try {
      const window = await activeWindow();
      
      if (!window) return;
      
      const appName = window.owner.name;
      const appTitle = window.title;
      
      // 检测应用切换
      if (appName !== this.currentApp) {
        this.onAppSwitch(appName);
        this.currentApp = appName;
      }
      
      // 累计使用时长
      const usage = this.appUsage.get(appName) || {
        name: appName,
        duration: 0,
        switchCount: 0,
        lastActive: Date.now(),
        windows: [],
      };
      
      usage.duration += 1; // 增加1秒
      usage.lastActive = Date.now();
      
      // 记录窗口标题（用于分析工作内容）
      if (!usage.windows.includes(appTitle)) {
        usage.windows.push(appTitle);
      }
      
      this.appUsage.set(appName, usage);
      
    } catch (error) {
      console.error('Failed to get active window:', error);
    }
  }
  
  private onAppSwitch(newApp: string): void {
    const usage = this.appUsage.get(newApp) || {
      name: newApp,
      duration: 0,
      switchCount: 0,
      lastActive: Date.now(),
      windows: [],
    };
    
    usage.switchCount++;
    this.appUsage.set(newApp, usage);
  }
  
  getStats(): ApplicationStats {
    const apps = Array.from(this.appUsage.values())
      .sort((a, b) => b.duration - a.duration);
    
    const totalDuration = apps.reduce((sum, app) => sum + app.duration, 0);
    const totalSwitches = apps.reduce((sum, app) => sum + app.switchCount, 0);
    
    return {
      totalApps: apps.length,
      totalSwitches,
      topApps: apps.slice(0, 10).map(app => ({
        name: app.name,
        duration: app.duration,
        durationFormatted: this.formatDuration(app.duration),
        percentage: ((app.duration / totalDuration) * 100).toFixed(1),
        switchCount: app.switchCount,
      })),
      currentApp: this.currentApp || 'Unknown',
    };
  }
  
  reset(): void {
    this.appUsage.clear();
  }
  
  private formatDuration(seconds: number): string {
    const hours = Math.floor(seconds / 3600);
    const minutes = Math.floor((seconds % 3600) / 60);
    const secs = seconds % 60;
    
    if (hours > 0) {
      return `${hours}h ${minutes}m`;
    } else if (minutes > 0) {
      return `${minutes}m ${secs}s`;
    } else {
      return `${secs}s`;
    }
  }
}

interface AppUsageData {
  name: string;
  duration: number; // 秒
  switchCount: number;
  lastActive: number;
  windows: string[];
}

interface ApplicationStats {
  totalApps: number;
  totalSwitches: number;
  topApps: Array<{
    name: string;
    duration: number;
    durationFormatted: string;
    percentage: string;
    switchCount: number;
  }>;
  currentApp: string;
}
```

```typescript
class ScreenshotTracker {
  private screenshotCount = 0;
  private screenshotTimes: number[] = [];
  
  start(): void {
    // 监听系统截图快捷键
    // Windows: Win+Shift+S, PrtScn
    // 这需要全局快捷键监听
    
    const { globalShortcut } = require('electron');
    
    // 注册截图快捷键监听
    globalShortcut.register('PrintScreen', () => {
      this.onScreenshot();
    });
    
    // 也可以监听截图目录的文件变化
    this.watchScreenshotFolder();
  }
  
  stop(): void {
    const { globalShortcut } = require('electron');
    globalShortcut.unregisterAll();
  }
  
  private watchScreenshotFolder(): void {
    const fs = require('fs');
    const path = require('path');
    const os = require('os');
    
    // Windows默认截图文件夹
    const screenshotPath = path.join(
      os.homedir(),
      'Pictures',
      'Screenshots'
    );
    
    if (fs.existsSync(screenshotPath)) {
      fs.watch(screenshotPath, (eventType: string, filename: string) => {
        if (eventType === 'rename' && filename) {
          // 新文件创建
          this.onScreenshot();
        }
      });
    }
  }
  
  private onScreenshot(): void {
    this.screenshotCount++;
    this.screenshotTimes.push(Date.now());
    
    // 只保留最近100次记录
    if (this.screenshotTimes.length > 100) {
      this.screenshotTimes.shift();
    }
    
    this.notifyUpdate();
  }
  
  getStats(): ScreenshotStats {
    return {
      total: this.screenshotCount,
      recent: this.screenshotTimes.slice(-10),
      averagePerHour: this.calculateAveragePerHour(),
    };
  }
  
  reset(): void {
    this.screenshotCount = 0;
    this.screenshotTimes = [];
  }
  
  private calculateAveragePerHour(): number {
    if (this.screenshotTimes.length < 2) return 0;
    
    const timeSpan = Date.now() - this.screenshotTimes[0];
    const hours = timeSpan / (1000 * 60 * 60);
    
    return Math.round(this.screenshotCount / hours);
  }
  
  private notifyUpdate(): void {
    BrowserWindow.getAllWindows()[0]?.webContents.send(
      'activity:update',
      { type: 'screenshot', data: this.getStats() }
    );
  }
}

interface ScreenshotStats {
  total: number;
  recent: number[];
  averagePerHour: number;
}
```

### 5. 剪贴板追踪器

```typescript
import { clipboard } from 'electron';

class ClipboardTracker {
  private copyCount = 0;
  private pasteCount = 0;
  private lastClipboard = '';
  private checkInterval: NodeJS.Timeout | null = null;
  
  start(): void {
    // 定期检查剪贴板变化（检测复制操作）
    this.checkInterval = setInterval(() => {
      this.checkClipboard();
    }, 500); // 每0.5秒检查一次
    
    // 监听粘贴快捷键（Ctrl+V）
    this.listenPasteShortcut();
  }
  
  stop(): void {
    if (this.checkInterval) {
      clearInterval(this.checkInterval);
    }
  }
  
  private checkClipboard(): void {
    const current = clipboard.readText();
    
    if (current !== this.lastClipboard && current.length > 0) {
      // 剪贴板内容变化，说明发生了复制操作
      this.copyCount++;
      this.lastClipboard = current;
      this.notifyUpdate();
    }
  }
  
  private listenPasteShortcut(): void {
    // 通过键盘监听检测 Ctrl+V
    // 简化实现，实际需要结合键盘追踪器
    const ioHook = require('iohook');
    
    let ctrlPressed = false;
    
    ioHook.on('keydown', (event: any) => {
      if (event.keycode === 29) { // Ctrl
        ctrlPressed = true;
      }
      if (event.keycode === 47 && ctrlPressed) { // V
        this.pasteCount++;
        this.notifyUpdate();
      }
    });
    
    ioHook.on('keyup', (event: any) => {
      if (event.keycode === 29) {
        ctrlPressed = false;
      }
    });
  }
  
  getStats(): ClipboardStats {
    return {
      copyCount: this.copyCount,
      pasteCount: this.pasteCount,
      ratio: this.pasteCount > 0 
        ? (this.copyCount / this.pasteCount).toFixed(2) 
        : '0',
    };
  }
  
  reset(): void {
    this.copyCount = 0;
    this.pasteCount = 0;
  }
  
  private notifyUpdate(): void {
    BrowserWindow.getAllWindows()[0]?.webContents.send(
      'activity:update',
      { type: 'clipboard', data: this.getStats() }
    );
  }
}

interface ClipboardStats {
  copyCount: number;
  pasteCount: number;
  ratio: string;
}
```

### 6. 活跃时间追踪器

```typescript
class ActivityTimeTracker {
  private activeTime = 0; // 活跃秒数
  private idleTime = 0; // 空闲秒数
  private lastActivityTime = Date.now();
  private isActive = true;
  private checkInterval: NodeJS.Timeout | null = null;
  
  private readonly IDLE_THRESHOLD = 5 * 60 * 1000; // 5分钟无操作视为空闲
  
  start(): void {
    // 每秒检查一次
    this.checkInterval = setInterval(() => {
      this.updateActivityStatus();
    }, 1000);
    
    // 监听用户活动（鼠标、键盘事件）
    this.listenUserActivity();
  }
  
  stop(): void {
    if (this.checkInterval) {
      clearInterval(this.checkInterval);
    }
  }
  
  private listenUserActivity(): void {
    const ioHook = require('iohook');
    
    // 任何鼠标或键盘活动都更新最后活动时间
    const updateActivity = () => {
      this.lastActivityTime = Date.now();
      if (!this.isActive) {
        this.isActive = true;
        this.notifyStatusChange('active');
      }
    };
    
    ioHook.on('mousemove', updateActivity);
    ioHook.on('mouseclick', updateActivity);
    ioHook.on('keydown', updateActivity);
  }
  
  private updateActivityStatus(): void {
    const now = Date.now();
    const timeSinceLastActivity = now - this.lastActivityTime;
    
    if (timeSinceLastActivity > this.IDLE_THRESHOLD) {
      // 空闲状态
      if (this.isActive) {
        this.isActive = false;
        this.notifyStatusChange('idle');
      }
      this.idleTime++;
    } else {
      // 活跃状态
      this.activeTime++;
    }
  }
  
  getStats(): ActivityTimeStats {
    const total = this.activeTime + this.idleTime;
    
    return {
      activeTime: this.activeTime,
      activeTimeFormatted: this.formatTime(this.activeTime),
      idleTime: this.idleTime,
      idleTimeFormatted: this.formatTime(this.idleTime),
      totalTime: total,
      totalTimeFormatted: this.formatTime(total),
      activePercentage: total > 0 
        ? ((this.activeTime / total) * 100).toFixed(1) 
        : '0',
      currentStatus: this.isActive ? 'active' : 'idle',
    };
  }
  
  reset(): void {
    this.activeTime = 0;
    this.idleTime = 0;
  }
  
  private formatTime(seconds: number): string {
    const hours = Math.floor(seconds / 3600);
    const minutes = Math.floor((seconds % 3600) / 60);
    const secs = seconds % 60;
    
    return `${hours.toString().padStart(2, '0')}:${minutes.toString().padStart(2, '0')}:${secs.toString().padStart(2, '0')}`;
  }
  
  private notifyStatusChange(status: 'active' | 'idle'): void {
    BrowserWindow.getAllWindows()[0]?.webContents.send(
      'activity:statusChange',
      status
    );
    
    // 更新西柚角色状态
    if (status === 'idle') {
      // 西柚进入休息/睡眠状态
    } else {
      // 西柚恢复活跃状态
    }
  }
  
  private notifyUpdate(): void {
    BrowserWindow.getAllWindows()[0]?.webContents.send(
      'activity:update',
      { type: 'time', data: this.getStats() }
    );
  }
}

interface ActivityTimeStats {
  activeTime: number;
  activeTimeFormatted: string;
  idleTime: number;
  idleTimeFormatted: string;
  totalTime: number;
  totalTimeFormatted: string;
  activePercentage: string;
  currentStatus: 'active' | 'idle';
}
```

```typescript
class ActivityStorageManager {
  private currentData: DailyActivityData | null = null;
  private saveInterval: NodeJS.Timeout | null = null;
  
  initialize(): void {
    // 加载今日数据
    this.loadTodayData();
    
    // 每5分钟自动保存一次
    this.saveInterval = setInterval(() => {
      this.saveCurrentData();
    }, 5 * 60 * 1000);
    
    // 每天凌晨归档昨日数据
    this.scheduleDailyArchive();
  }
  
  updateCurrentData(type: string, data: any): void {
    if (!this.currentData) return;
    
    // 根据类型更新对应字段
    switch (type) {
      case 'mouse':
        this.currentData.summary.mouseClicks = data.clicks;
        this.currentData.summary.mouseDistance = data.moveDistance;
        break;
      case 'keyboard':
        this.currentData.summary.keyPresses = data.totalKeys;
        this.currentData.summary.characters = data.characters;
        break;
      // ... 其他类型
    }
    
    // 更新每小时数据
    this.updateHourlyData();
  }
  
  private loadTodayData(): void {
    const today = this.getTodayString();
    const filePath = this.getDataPath(today);
    
    if (fs.existsSync(filePath)) {
      this.currentData = JSON.parse(fs.readFileSync(filePath, 'utf-8'));
    } else {
      this.currentData = this.createEmptyData(today);
    }
  }
  
  private saveCurrentData(): void {
    if (!this.currentData) return;
    
    const filePath = this.getDataPath(this.currentData.date);
    
    // 保存到文件
    fs.writeFileSync(
      filePath,
      JSON.stringify(this.currentData, null, 2),
      'utf-8'
    );
    
    // 同时保存到 current.json
    fs.writeFileSync(
      path.join(this.getActivityDir(), 'current.json'),
      JSON.stringify(this.currentData, null, 2),
      'utf-8'
    );
  }
  
  private scheduleDailyArchive(): void {
    // 计算到明天凌晨的时间
    const now = new Date();
    const tomorrow = new Date(now);
    tomorrow.setDate(tomorrow.getDate() + 1);
    tomorrow.setHours(0, 0, 0, 0);
    
    const timeUntilMidnight = tomorrow.getTime() - now.getTime();
    
    setTimeout(() => {
      this.archiveYesterdayData();
      
      // 然后每24小时执行一次
      setInterval(() => {
        this.archiveYesterdayData();
      }, 24 * 60 * 60 * 1000);
    }, timeUntilMidnight);
  }
  
  private archiveYesterdayData(): void {
    // 保存昨日数据
    this.saveCurrentData();
    
    // 生成昨日统计
    this.generateDailySummary();
    
    // 检查是否需要生成周/月统计
    this.checkAndGenerateWeeklyMonthly();
    
    // 创建新的今日数据
    this.currentData = this.createEmptyData(this.getTodayString());
  }
  
  private updateHourlyData(): void {
    if (!this.currentData) return;
    
    const currentHour = new Date().getHours();
    
    // 找到或创建当前小时的数据
    let hourData = this.currentData.hourlyActivity.find(h => h.hour === currentHour);
    
    if (!hourData) {
      hourData = {
        hour: currentHour,
        mouseClicks: 0,
        keyPresses: 0,
        activeMinutes: 0,
      };
      this.currentData.hourlyActivity.push(hourData);
    }
    
    // 更新数据
    hourData.mouseClicks = this.currentData.summary.mouseClicks.total;
    hourData.keyPresses = this.currentData.summary.keyPresses;
    // ...
  }
  
  // 查询历史数据
  async getHistoryData(startDate: string, endDate: string): Promise<DailyActivityData[]> {
    const data: DailyActivityData[] = [];
    
    // 遍历日期范围
    const start = new Date(startDate);
    const end = new Date(endDate);
    
    for (let date = start; date <= end; date.setDate(date.getDate() + 1)) {
      const dateStr = this.formatDate(date);
      const filePath = this.getDataPath(dateStr);
      
      if (fs.existsSync(filePath)) {
        const dayData = JSON.parse(fs.readFileSync(filePath, 'utf-8'));
        data.push(dayData);
      }
    }
    
    return data;
  }
  
  private getTodayString(): string {
    return this.formatDate(new Date());
  }
  
  private formatDate(date: Date): string {
    return date.toISOString().split('T')[0];
  }
  
  private getDataPath(date: string): string {
    const [year, month] = date.split('-');
    const dir = path.join(this.getActivityDir(), year, month);
    
    // 确保目录存在
    fs.mkdirSync(dir, { recursive: true });
    
    return path.join(dir, `${date}.json`);
  }
  
  private getActivityDir(): string {
    return path.join(app.getPath('userData'), 'activity');
  }
  
  private createEmptyData(date: string): DailyActivityData {
    return {
      date,
      summary: {
        mouseClicks: { total: 0, left: 0, right: 0, middle: 0 },
        mouseDistance: 0,
        keyPresses: 0,
        characters: 0,
        wordsEstimate: 0,
        activeTime: 0,
        idleTime: 0,
        startTime: Date.now(),
        endTime: 0,
        screenshots: 0,
        copyPaste: { copy: 0, paste: 0 },
        appSwitches: 0,
      },
      applications: [],
      hourlyActivity: [],
      metadata: {
        version: '1.0.0',
        platform: process.platform,
        recordedBy: '西柚桌宠',
      },
    };
  }
}
```

## 🔒 隐私与安全

### 隐私保护原则

1. **完全本地存储**：所有数据只存储在用户设备上
2. **不记录敏感内容**：
   - 不记录按键内容（只统计次数）
   - 不记录屏幕内容
   - 不记录剪贴板内容（只统计次数）
   - 应用窗口标题可选择性记录
3. **用户完全控制**：
   - 可随时暂停/停止监控
   - 可删除任何历史数据
   - 可导出/备份数据
   - 可配置监控范围

## ⚠️ 注意事项

### 1. 性能影响

全局监听可能影响系统性能：
- 使用节流（throttle）限制更新频率
- 批量处理事件
- 异步写入磁盘
- 避免在主进程做复杂计算

### 2. 权限要求

某些功能需要系统权限：
- **Windows**：需要管理员权限（iohook）
- **Mac**：需要辅助功能权限
- 首次启动时引导用户授权

### 3. 跨平台兼容

```typescript
// 根据平台选择不同实现
const createTracker = () => {
  if (process.platform === 'win32') {
    return new WindowsTracker();
  } else if (process.platform === 'darwin') {
    return new MacTracker();
  } else {
    return new LinuxTracker();
  }
};
```

### 4. 用户透明度

- 首次启动时明确告知用户功能
- 提供详细的隐私说明
- 允许用户随时查看和删除数据
- 在任务栏图标显示监控状态
