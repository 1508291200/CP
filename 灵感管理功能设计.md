#### 3.1 自动分类算法

**多层级分类策略**

```typescript
class InspirationClassifier {
  /**
   * 自动分类灵感
   * 综合使用关键词匹配、规则引擎、上下文分析
   */
  classify(inspiration: Partial<Inspiration>): InspirationCategory {
    // 1. 关键词匹配（70%权重）
    const keywordScore = this.analyzeKeywords(inspiration.content);
    
    // 2. 标签分析（20%权重）
    const tagScore = this.analyzeTags(inspiration.tags);
    
    // 3. 上下文分析（10%权重）
    const contextScore = this.analyzeContext(inspiration.context);
    
    // 综合评分
    return this.selectCategory(keywordScore, tagScore, contextScore);
  }
  
  private analyzeKeywords(content: string): CategoryScores {
    const scores: CategoryScores = {};
    
    // 关键词字典
    const keywords = {
      work: ['会议', '任务', '工作', '项目进度', '汇报', 'deadline'],
      project: ['想做', '开发', '项目', '功能', '产品', '实现'],
      learning: ['学习', '笔记', '理解', '掌握', '教程', '知识点'],
      technical: ['代码', '算法', '架构', '优化', 'bug', '方案'],
      design: ['设计', 'UI', 'UX', '界面', '交互', '视觉', '配色'],
      creative: ['创意', '点子', '灵感', '想法', '脑洞'],
      // ...
    };
    
    // 计算每个分类的匹配分数
    Object.entries(keywords).forEach(([category, words]) => {
      scores[category] = words.reduce((score, word) => {
        return score + (content.includes(word) ? 1 : 0);
      }, 0);
    });
    
    return scores;
  }
  
  private analyzeTags(tags?: string[]): CategoryScores {
    // 基于已有标签推断分类
    if (!tags || tags.length === 0) return {};
    
    // 标签-分类映射
    const tagCategoryMap = {
      'JavaScript': 'technical',
      'Python': 'technical',
      'UI设计': 'design',
      '商业模式': 'business',
      // ...从历史数据学习
    };
    
    const scores: CategoryScores = {};
    tags.forEach(tag => {
      const category = tagCategoryMap[tag];
      if (category) {
        scores[category] = (scores[category] || 0) + 2;
      }
    });
    
    return scores;
  }
}
```

#### 3.2 智能标签提取

**自动提取关键词作为标签**

```typescript
class TagExtractor {
  /**
   * 从内容中提取标签
   */
  extract(content: string): string[] {
    const tags: string[] = [];
    
    // 1. 识别已有的 #标签
    const hashTags = content.match(/#[\u4e00-\u9fa5\w]+/g);
    if (hashTags) {
      tags.push(...hashTags.map(t => t.slice(1)));
    }
    
    // 2. 提取技术关键词
    const techKeywords = this.extractTechKeywords(content);
    tags.push(...techKeywords);
    
    // 3. 提取高频名词
    const nouns = this.extractNouns(content);
    tags.push(...nouns.slice(0, 5)); // 最多5个
    
    // 去重并返回
    return [...new Set(tags)];
  }
  
  private extractTechKeywords(content: string): string[] {
    // 技术关键词词典
    const techKeywords = [
      'JavaScript', 'Python', 'React', 'Vue', 'Node.js',
      'AI', 'ML', '机器学习', '深度学习',
      '前端', '后端', '全栈', '算法', '数据结构',
      // ...
    ];
    
    return techKeywords.filter(keyword => 
      content.toLowerCase().includes(keyword.toLowerCase())
    );
  }
  
  private extractNouns(content: string): string[] {
    // 简单的中文分词（实际可用第三方库如 nodejieba）
    // 这里用简化版本
    
    // 常见名词模式
    const nounPatterns = [
      /[\u4e00-\u9fa5]{2,4}(?=的|了|是|在)/g,
    ];
    
    const nouns: string[] = [];
    nounPatterns.forEach(pattern => {
      const matches = content.match(pattern);
      if (matches) {
        nouns.push(...matches);
      }
    });
    
    // 频率统计，返回高频词
    const frequency = this.countFrequency(nouns);
    return Object.keys(frequency)
      .sort((a, b) => frequency[b] - frequency[a])
      .slice(0, 5);
  }
  
  private countFrequency(words: string[]): Record<string, number> {
    return words.reduce((acc, word) => {
      acc[word] = (acc[word] || 0) + 1;
      return acc;
    }, {} as Record<string, number>);
  }
}
```

```typescript
class InspirationSearchEngine {
  search(query: string, inspirations: Inspiration[]): Inspiration[] {
    const parser = new SearchQueryParser(query);
    const filters = parser.parse();
    
    return inspirations.filter(inspiration => {
      // 全文搜索
      if (filters.text && !this.matchText(inspiration, filters.text)) {
        return false;
      }
      
      // 标签过滤
      if (filters.tags && !this.matchTags(inspiration, filters.tags)) {
        return false;
      }
      
      // 分类过滤
      if (filters.category && inspiration.category !== filters.category) {
        return false;
      }
      
      // 日期过滤
      if (filters.dateRange && !this.matchDateRange(inspiration, filters.dateRange)) {
        return false;
      }
      
      // 状态过滤
      if (filters.status && !this.matchStatus(inspiration, filters.status)) {
        return false;
      }
      
      return true;
    });
  }
  
  private matchText(inspiration: Inspiration, text: string): boolean {
    const searchableText = [
      inspiration.title,
      inspiration.content,
      ...inspiration.tags,
    ].join(' ').toLowerCase();
    
    return searchableText.includes(text.toLowerCase());
  }
}
```

```typescript
class InspirationRecommender {
  /**
   * 推荐相关灵感
   */
  recommend(currentId: string, allInspirations: Inspiration[]): Inspiration[] {
    const current = allInspirations.find(i => i.id === currentId);
    if (!current) return [];
    
    // 计算相似度评分
    const scored = allInspirations
      .filter(i => i.id !== currentId)
      .map(inspiration => ({
        inspiration,
        score: this.calculateSimilarity(current, inspiration),
      }))
      .sort((a, b) => b.score - a.score);
    
    // 返回前5个
    return scored.slice(0, 5).map(s => s.inspiration);
  }
  
  private calculateSimilarity(a: Inspiration, b: Inspiration): number {
    let score = 0;
    
    // 相同分类 +10分
    if (a.category === b.category) {
      score += 10;
    }
    
    // 相同标签，每个 +5分
    const commonTags = a.tags.filter(tag => b.tags.includes(tag));
    score += commonTags.length * 5;
    
    // 时间接近 +3分
    const daysDiff = Math.abs(a.createdAt - b.createdAt) / (1000 * 60 * 60 * 24);
    if (daysDiff < 7) {
      score += 3;
    }
    
    // 内容相似度（简化版）
    const contentSimilarity = this.calculateTextSimilarity(a.content, b.content);
    score += contentSimilarity * 20;
    
    return score;
  }
  
  private calculateTextSimilarity(text1: string, text2: string): number {
    // 简化版：基于共同词汇
    const words1 = new Set(text1.split(/\s+/));
    const words2 = new Set(text2.split(/\s+/));
    
    const intersection = new Set([...words1].filter(w => words2.has(w)));
    const union = new Set([...words1, ...words2]);
    
    return intersection.size / union.size; // Jaccard相似度
  }
}
```

#### 6.1 "今日回顾"

```typescript
class DailyReview {
  /**
   * 每日回顾：从历史灵感中随机选择一条
   */
  getDailyReview(): Inspiration | null {
    const allInspirations = this.loadAllInspirations();
    
    // 过滤条件：
    // 1. 至少7天前
    // 2. 未归档
    // 3. 未在最近7天内回顾过
    
    const candidates = allInspirations.filter(i => {
      const age = Date.now() - i.createdAt;
      const sevenDays = 7 * 24 * 60 * 60 * 1000;
      
      return (
        age > sevenDays &&
        !i.archived &&
        !this.wasRecentlyReviewed(i.id)
      );
    });
    
    if (candidates.length === 0) return null;
    
    // 加权随机（优先级高的更容易被选中）
    return this.weightedRandomSelect(candidates);
  }
  
  private weightedRandomSelect(inspirations: Inspiration[]): Inspiration {
    const weights = inspirations.map(i => i.priority || 3);
    const totalWeight = weights.reduce((a, b) => a + b, 0);
    
    let random = Math.random() * totalWeight;
    for (let i = 0; i < inspirations.length; i++) {
      random -= weights[i];
      if (random <= 0) {
        return inspirations[i];
      }
    }
    
    return inspirations[0];
  }
}
```
